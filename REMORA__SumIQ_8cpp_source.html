121REMORA::volWgtSumMF(int lev, const MultiFab& mf, int comp, bool local, bool finemask)
122{
123 BL_PROFILE("REMORA::volWgtSumMF()");

125 Real sum = 0.0_rt;

146 vol_arr(i,j,k) = Hz(i,j,k) / (pm(i,j,0) * pn(i,j,0));
