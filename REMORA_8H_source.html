1484 /** \brief Evaluate stability function psi for wind speed */
1485 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
1486 static amrex::Real bulk_psiu(amrex::Real ZoL) {
1487 // Compute stability function, psi
1488 using namespace amrex;
1489
1490 Real psiu;
1491 if (ZoL < 0.0_rt) {
1492 // Unstable conditions.
1493 Real x = std::pow(1.0_rt-15.0_rt*ZoL,0.25_rt);
1494 Real psik = 2.0_rt*std::log(0.5_rt*(1.0_rt+x))+
1495 std::log(0.5_rt*(1.0_rt+x*x))-
1496 2.0_rt*std::atan(x)+0.5_rt*PI;
1497 // For very unstable conditions, use free-convection (Fairall).
1498 Real sqrt3 = std::sqrt(3.0_rt);
1499 Real y = std::pow(1.0_rt-10.15_rt*ZoL,1.0_rt/3.0_rt);
1500 Real psic = 1.5_rt*std::log(1.0_rt/3.0_rt*(1.0_rt+y+y*y))-
1501 sqrt3*std::atan((1.0_rt+2.0_rt*y)/sqrt3)+PI/sqrt3;
1502 // Match Kansas and free-convection forms with weighting Fw.
1503 Real Zol2 = ZoL*ZoL;
1504 Real Fw = Zol2/(1.0_rt+Zol2);
1505
1506 psiu = (1.0_rt-Fw)*psik+Fw*psic;
1507 } else {
1508 // Stable conditions
1509 Real cff=std::min(50.0_rt,0.35_rt*ZoL);
1510 psiu = -((1.0_rt+ZoL)+0.6667_rt*(ZoL-14.28_rt)/
1511 std::exp(cff)+8.525_rt);
1512 }
1513 return psiu;
1514 }
