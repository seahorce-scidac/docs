35Geometry
36make_unit_slab_geometry (Box const& domain)
37{
38 static constexpr Real lo[AMREX_SPACEDIM] = {0.0_rt, 0.0_rt, 0.0_rt};

48 Vector<BCRec> bcs(ncomp);

50 for (int dir = 0; dir < AMREX_SPACEDIM; ++dir) {

84amrex::Real
85REMORA::EvolveOneStep (amrex::Real /*time*/, amrex::Real /*dt_request*/)
86{
87 Real cur_time = t_new[0];
88 const int step = istep[0];
89
90 if (cur_time >= stop_time) {
91 return Real(0.0);
92 }
93
94 ComputeDt();
95
96 int lev = 0;
97 int iteration = 1;
98 if (max_level == 0) {
99 timeStep(lev, cur_time, iteration);
100 } else {
101 timeStepML(cur_time, iteration);
102 }
103
104 cur_time += dt[0];
105
106 WriteAtIntermediateTime(step, cur_time);
107
108 post_timestep(step, cur_time, dt[0]);
109
110 return dt[0];
111}

162/*
163 * \brief Extracts SST from the 3D conservative state for the atmospheric driver.
164 *
165 * Reads Temp_comp at the top water-column cell (k_sfc), converts from
166 * Celsius to Kelvin, and copies the result into state[SSTIndex].
167 *
168 * @param[in,out] state OCN2ATM slab buffer sized by the driver (one MultiFab
169 * per ocean-to-atmosphere export layer). state[SSTIndex]
170 * (index 0) is overwritten with sea-surface temperature
171 * (SST) sampled from the Temp_comp tracer at the uppermost
172 * sigma level (k = Nz) and converted from degrees Celsius
173 * to Kelvin for the atmospheric driver. An empty vector or
174 * null state[0] is treated as a no-op.
175 * @param[in ] time Current ocean model time (unused; retained for driver
176 * interface conformance).
177 */
178void
179REMORA::PackSurfaceState (Vector<MultiFab*>& state, Real /*time*/)
180{
181 if (state.empty() || state[SSTIndex] == nullptr) { return; }
182 const int lev = 0;
183
184 // REMORA stores temperature in Celsius. Surface is at k=N (top of water column).
185 const int k_sfc = cons_new[lev]->boxArray().minimalBox().bigEnd(2);
186
187 // Build a temp MultiFab on REMORA's ba2d (k=0) derived from cons_new's BoxArray.
188 // Same DistributionMap ensures each box is local; we fill at k=0 from cons at k=k_sfc.
189 BoxList bl2d = cons_new[lev]->boxArray().boxList();
190 for (auto& b : bl2d) { b.setRange(2, 0); }
191 BoxArray ba2d(std::move(bl2d));
192 MultiFab tmp(ba2d, cons_new[lev]->DistributionMap(), 1, 0);
193
194 for (MFIter mfi(*cons_new[lev]); mfi.isValid(); ++mfi) {
195 auto const& c = cons_new[lev]->const_array(mfi);
196 auto t = tmp.array(mfi);
197 Box bx = makeSlab(mfi.validbox(), 2, k_sfc);
198 ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int) {
199 // Write to k=0 in tmp (ba2d range); convert Celsius → Kelvin.
200 t(i, j, 0) = c(i, j, k_sfc, Temp_comp) + 273.15_rt;
201 });
202 }
203
204 MultiFab& dst = *state[SSTIndex];
205 const Box src_domain = tmp.boxArray().minimalBox();
206 const Box dst_domain = dst.boxArray().minimalBox();
207
208 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(
209 src_domain.smallEnd(2) == src_domain.bigEnd(2) &&
210 dst_domain.smallEnd(2) == dst_domain.bigEnd(2),
211 "REMORA::PackSurfaceState expects 2D slab source and destination MultiFabs.");
212 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(
213 src_domain.smallEnd(0) == dst_domain.smallEnd(0) &&
214 src_domain.smallEnd(1) == dst_domain.smallEnd(1),
215 "REMORA::PackSurfaceState requires aligned atmosphere/ocean slab origins.");
216
217 const IntVect src_len = src_domain.length();
218 const IntVect dst_len = dst_domain.length();
219 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(
220 src_len[2] == 1 && dst_len[2] == 1,
221 "REMORA::PackSurfaceState expects unit-thickness source and destination slabs.");
222
223 const bool same_xy = (src_len[0] == dst_len[0] && src_len[1] == dst_len[1]);
224 const bool dst_finer =
225 (dst_len[0] >= src_len[0] && dst_len[1] >= src_len[1] &&
226 dst_len[0] % src_len[0] == 0 && dst_len[1] % src_len[1] == 0);
227 const bool dst_coarser =
228 (src_len[0] >= dst_len[0] && src_len[1] >= dst_len[1] &&
229 src_len[0] % dst_len[0] == 0 && src_len[1] % dst_len[1] == 0);
230
231 if (same_xy) {
232 dst.ParallelCopy(tmp, 0, 0, 1);
233 return;
234 }
235
236 if (dst_finer) {
237 const IntVect ratio(dst_len[0] / src_len[0], dst_len[1] / src_len[1], 1);
238 const auto src_geom = make_unit_slab_geometry(src_domain);
239 const auto dst_geom = make_unit_slab_geometry(dst_domain);
240 const auto bcs = make_internal_bcs(1);
241 amrex::InterpFromCoarseLevel(dst, IntVect(0), IntVect(0),
242 tmp, 0, 0, 1,
243 src_geom, dst_geom,
244 ratio, &pc_interp, bcs, 0);
245 return;
246 }
247
248 if (dst_coarser) {
249 const IntVect ratio(src_len[0] / dst_len[0], src_len[1] / dst_len[1], 1);
250 amrex::average_down(tmp, dst, 0, 1, ratio);
251 return;
252 }
253
254 amrex::Abort("REMORA::PackSurfaceState requires matching horizontal extents and integer-ratio atmosphere/ocean slab resolutions.");
255}

257/*
258 * \brief Receives atmospheric states from the driver and applies unit conversions.
259 *
260 * Fills REMORA's internal forcing MultiFabs from states and records which
261 * lanes were successfully updated in driver_atmos_state_from_driver.
262 * Unit conversions applied: Pair Pa to mb; Tair K to Celsius.
263 *
264 * @param[in] states ATM2OCN forcing slab buffer from the driver (Warner et al.
265 * 2010, Block B state-passing contract), indexed by AtmosState.
266 * Expected units per lane:
267 * Uwind/Vwind: 10-m winds [m/s];
268 * Pair: mean sea-level pressure [Pa, converted to mb];
269 * Qair: near-surface specific humidity [kg/kg];
270 * Tair: 2-m air temperature [K, converted to degC];
271 * Cloud: cloud fraction [0-1];
272 * Rain: precipitation rate [kg/m^2/s];
273 * SWrad/LWrad: downwelling shortwave/longwave radiation [W/m^2].
274 * Missing lanes (null pointer or index out of range) are skipped;
275 * driver_atmos_state_from_driver tracks populated lanes for the
276 * bulk-flux parameterization fallback logic.
277 * @param[in] time Current ocean model time (unused; retained for driver
278 * interface conformance).
279 */
280void
281REMORA::ApplyAtmosphericStates (const Vector<MultiFab*>& states, Real /*time*/)
282{
283 running_with_coupling_driver = true;
284 driver_atmos_forcing_mode = DriverAtmosForcingMode::State;
285 driver_atmos_state_from_driver.fill(false);
286 if (finest_level < 0) { return; }
287
288 // Wind (m/s) — no unit conversion
289 if (vec_uwind[0] != nullptr) {
290 if (states.size() > AtmosState::Uwind && states[AtmosState::Uwind] != nullptr) {
291 vec_uwind[0]->ParallelCopy(*states[AtmosState::Uwind], 0, 0, 1);
292 vec_uwind[0]->FillBoundary(geom[0].periodicity());
293 driver_atmos_state_from_driver[AtmosState::Uwind] = true;
294 }
295 }
296 if (vec_vwind[0] != nullptr) {
297 if (states.size() > AtmosState::Vwind && states[AtmosState::Vwind] != nullptr) {
298 vec_vwind[0]->ParallelCopy(*states[AtmosState::Vwind], 0, 0, 1);
299 vec_vwind[0]->FillBoundary(geom[0].periodicity());
300 driver_atmos_state_from_driver[AtmosState::Vwind] = true;
301 }
302 }
303
304 // Atmospheric pressure: Pa → mb (REMORA bulk flux expects mb)
305 if (vec_Pair[0] != nullptr) {
306 if (states.size() > AtmosState::Pair && states[AtmosState::Pair] != nullptr) {
307 vec_Pair[0]->ParallelCopy(*states[AtmosState::Pair], 0, 0, 1);
308 vec_Pair[0]->mult(0.01_rt, 0, 1);
309 vec_Pair[0]->FillBoundary(geom[0].periodicity());
310 driver_atmos_state_from_driver[AtmosState::Pair] = true;
311 }
312 }
313
314 // Specific humidity (kg/kg) — no conversion
315 if (vec_qair[0] != nullptr) {
316 if (states.size() > AtmosState::Qair && states[AtmosState::Qair] != nullptr) {
317 vec_qair[0]->ParallelCopy(*states[AtmosState::Qair], 0, 0, 1);
318 vec_qair[0]->FillBoundary(geom[0].periodicity());
319 driver_atmos_state_from_driver[AtmosState::Qair] = true;
320 }
321 }
322
323 // Air temperature: K → °C (REMORA stores/uses Celsius internally)
324 if (vec_Tair[0] != nullptr) {
325 if (states.size() > AtmosState::Tair && states[AtmosState::Tair] != nullptr) {
326 vec_Tair[0]->ParallelCopy(*states[AtmosState::Tair], 0, 0, 1);
327 vec_Tair[0]->plus(-273.15_rt, 0, 1);
328 vec_Tair[0]->FillBoundary(geom[0].periodicity());
329 driver_atmos_state_from_driver[AtmosState::Tair] = true;
330 }
331 }
332
333 // Cloud fraction [0-1], rain, SW/LW radiation — no unit conversion
334 if (vec_cloud[0] != nullptr) {
335 if (states.size() > AtmosState::Cloud && states[AtmosState::Cloud] != nullptr) {
336 vec_cloud[0]->ParallelCopy(*states[AtmosState::Cloud], 0, 0, 1);
337 vec_cloud[0]->FillBoundary(geom[0].periodicity());
338 driver_atmos_state_from_driver[AtmosState::Cloud] = true;
339 }
340 }
341 if (vec_rain[0] != nullptr) {
342 if (states.size() > AtmosState::Rain && states[AtmosState::Rain] != nullptr) {
343 vec_rain[0]->ParallelCopy(*states[AtmosState::Rain], 0, 0, 1);
344 vec_rain[0]->FillBoundary(geom[0].periodicity());
345 driver_atmos_state_from_driver[AtmosState::Rain] = true;
346 }
347 }
348 if (vec_srflx[0] != nullptr) {
349 if (states.size() > AtmosState::SWrad && states[AtmosState::SWrad] != nullptr) {
350 vec_srflx[0]->ParallelCopy(*states[AtmosState::SWrad], 0, 0, 1);
351 vec_srflx[0]->FillBoundary(geom[0].periodicity());
352 driver_atmos_state_from_driver[AtmosState::SWrad] = true;
353 }
354 }
355 if (vec_longwave_down[0] != nullptr) {
356 if (states.size() > AtmosState::LWrad && states[AtmosState::LWrad] != nullptr) {
357 vec_longwave_down[0]->ParallelCopy(*states[AtmosState::LWrad], 0, 0, 1);
358 vec_longwave_down[0]->FillBoundary(geom[0].periodicity());
359 driver_atmos_state_from_driver[AtmosState::LWrad] = true;
360 }
361 }

365void
366REMORA::ApplyAtmosphericFluxes (const Vector<MultiFab*>& states, Real /*time*/)
367{
368 running_with_coupling_driver = true;
369 driver_atmos_forcing_mode = DriverAtmosForcingMode::Flux;
370 driver_atmos_state_from_driver.fill(false);
371 if (finest_level < 0) { return; }
372
373 if (states.size() <= AtmosFluxes::Evap ||
374 states[AtmosFluxes::TauX] == nullptr ||
375 states[AtmosFluxes::TauY] == nullptr ||
376 states[AtmosFluxes::SHflux] == nullptr ||
377 states[AtmosFluxes::LHflux] == nullptr ||
378 states[AtmosFluxes::SWrad] == nullptr ||
379 states[AtmosFluxes::LWrad] == nullptr ||
380 states[AtmosFluxes::Rain] == nullptr ||
381 states[AtmosFluxes::Evap] == nullptr ||
382 vec_sustr[0] == nullptr || vec_svstr[0] == nullptr ||
383 vec_stflux[0] == nullptr || vec_mskr[0] == nullptr ||
384 vec_msku[0] == nullptr || vec_mskv[0] == nullptr ||
385 vec_srflx[0] == nullptr || vec_lrflx[0] == nullptr ||
386 vec_lhflx[0] == nullptr || vec_shflx[0] == nullptr ||
387 vec_rain[0] == nullptr || vec_evap[0] == nullptr) {
388 return;
389 }
390
391 const Real Hscale2 = 1.0_rt / (solverChoice.rho0 * Cp);
392 const Real rho0 = solverChoice.rho0;
393
394 MultiFab tau_x_tmp(vec_sustr[0]->boxArray(), vec_sustr[0]->DistributionMap(), 1,
395 vec_sustr[0]->nGrowVect());
396 MultiFab tau_y_tmp(vec_svstr[0]->boxArray(), vec_svstr[0]->DistributionMap(), 1,
397 vec_svstr[0]->nGrowVect());
398 MultiFab shflux_tmp(vec_shflx[0]->boxArray(), vec_shflx[0]->DistributionMap(), 1,
399 vec_shflx[0]->nGrowVect());
400 MultiFab lhflux_tmp(vec_lhflx[0]->boxArray(), vec_lhflx[0]->DistributionMap(), 1,
401 vec_lhflx[0]->nGrowVect());
402 MultiFab lwflux_tmp(vec_lrflx[0]->boxArray(), vec_lrflx[0]->DistributionMap(), 1,
403 vec_lrflx[0]->nGrowVect());
404
405 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::TauX], tau_x_tmp, geom[0],
406 "REMORA::ApplyAtmosphericFluxes expected TauX slab to match REMORA u-face layout.");
407 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::TauY], tau_y_tmp, geom[0],
408 "REMORA::ApplyAtmosphericFluxes expected TauY slab to match REMORA v-face layout.");
409 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::SHflux], shflux_tmp, geom[0],
410 "REMORA::ApplyAtmosphericFluxes expected SHflux slab to match REMORA rho layout.");
411 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::LHflux], lhflux_tmp, geom[0],
412 "REMORA::ApplyAtmosphericFluxes expected LHflux slab to match REMORA rho layout.");
413 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::LWrad], lwflux_tmp, geom[0],
414 "REMORA::ApplyAtmosphericFluxes expected LWrad slab to match REMORA rho layout.");
415 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::SWrad], *vec_srflx[0], geom[0],
416 "REMORA::ApplyAtmosphericFluxes expected SWrad slab to match REMORA rho layout.");
417 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::Rain], *vec_rain[0], geom[0],
418 "REMORA::ApplyAtmosphericFluxes expected rain slab to match REMORA rho layout.");
419 CopyDriverSlabToRemoraLayout(*states[AtmosFluxes::Evap], *vec_evap[0], geom[0],
420 "REMORA::ApplyAtmosphericFluxes expected evap slab to match REMORA rho layout.");
421 vec_lrflx[0]->setVal(0.0);
422 vec_lhflx[0]->setVal(0.0);
423 vec_shflx[0]->setVal(0.0);
424 vec_stflux[0]->setVal(0.0);
425
426 for (MFIter mfi(*vec_sustr[0], TilingIfNotGPU()); mfi.isValid(); ++mfi) {
427 Array4<Real> const& sustr = vec_sustr[0]->array(mfi);
428 Array4<const Real> const& msku = vec_msku[0]->const_array(mfi);
429 Array4<const Real> const& tau_x = tau_x_tmp.const_array(mfi);
430 Box ubx = mfi.grownnodaltilebox(0, IntVect(NGROW,NGROW,0));
431 Box ubxD = ubx;
432 ubxD.makeSlab(2,0);
433 ParallelFor(ubxD, [=] AMREX_GPU_DEVICE (int i, int j, int ) {
434 sustr(i,j,0) = tau_x(i,j,0) / rho0 * msku(i,j,0);
435 });
436 }
437
438 for (MFIter mfi(*vec_svstr[0], TilingIfNotGPU()); mfi.isValid(); ++mfi) {
439 Array4<Real> const& svstr = vec_svstr[0]->array(mfi);
440 Array4<const Real> const& mskv = vec_mskv[0]->const_array(mfi);
441 Array4<const Real> const& tau_y = tau_y_tmp.const_array(mfi);
442 Box vbx = mfi.grownnodaltilebox(1, IntVect(NGROW,NGROW,0));
443 Box vbxD = vbx;
444 vbxD.makeSlab(2,0);
445
446 ParallelFor(vbxD, [=] AMREX_GPU_DEVICE (int i, int j, int ) {
447 svstr(i,j,0) = tau_y(i,j,0) / rho0 * mskv(i,j,0);
448 });
449 }
450
451 for (MFIter mfi(*vec_stflux[0], TilingIfNotGPU()); mfi.isValid(); ++mfi) {
452 Array4<Real> const& stflux = vec_stflux[0]->array(mfi);
453 Array4<Real> const& lrflx = vec_lrflx[0]->array(mfi);
454 Array4<Real> const& lhflx = vec_lhflx[0]->array(mfi);
455 Array4<Real> const& shflx = vec_shflx[0]->array(mfi);
456 Array4<const Real> const& mskr = vec_mskr[0]->const_array(mfi);
457 Array4<const Real> const& srflx = vec_srflx[0]->const_array(mfi);
458 Array4<const Real> const& rain = vec_rain[0]->const_array(mfi);
459 Array4<const Real> const& evap = vec_evap[0]->const_array(mfi);
460 Array4<const Real> const& shflux = shflux_tmp.const_array(mfi);
461 Array4<const Real> const& lhflux = lhflux_tmp.const_array(mfi);
462 Array4<const Real> const& lwflux = lwflux_tmp.const_array(mfi);
463
464 Box gbx2 = mfi.growntilebox(IntVect(NGROW,NGROW,0));
465 Box gbx2D = gbx2;
466 gbx2D.makeSlab(2,0);
467
468 ParallelFor(gbx2D, [=] AMREX_GPU_DEVICE (int i, int j, int ) {
469 // ERF exports flux lanes in native surface-flux units; convert once at ingest.
470 lrflx(i,j,0) = lwflux(i,j,0) * Hscale2;
471 lhflx(i,j,0) = -lhflux(i,j,0) * Hscale2;
472 shflx(i,j,0) = -shflux(i,j,0) * Hscale2;
473 stflux(i,j,0,Temp_comp) =
474 (srflx(i,j,0) * Hscale2 + lrflx(i,j,0) + lhflx(i,j,0) + shflx(i,j,0))
475 * mskr(i,j,0);
476 stflux(i,j,0,Salt_comp) =
477 mskr(i,j,0) * (evap(i,j,0) - rain(i,j,0)) / rhow;
478 });
479 }
480
481 vec_sustr[0]->FillBoundary(geom[0].periodicity());
482 vec_svstr[0]->FillBoundary(geom[0].periodicity());
483 vec_srflx[0]->FillBoundary(geom[0].periodicity());
484 vec_lrflx[0]->FillBoundary(geom[0].periodicity());
485 vec_lhflx[0]->FillBoundary(geom[0].periodicity());
486 vec_shflx[0]->FillBoundary(geom[0].periodicity());
487 vec_stflux[0]->FillBoundary(geom[0].periodicity());
488 vec_rain[0]->FillBoundary(geom[0].periodicity());
489 vec_evap[0]->FillBoundary(geom[0].periodicity());
490 vec_stflux[0]->FillBoundary(geom[0].periodicity());
491
492 if (amrex::ParallelDescriptor::IOProcessor()) {
493 amrex::Print() << "REMORA ApplyAtmosphericFluxes validation:\n"
