23/**
24 * \brief NDArray is the datatype designed to hold any data, including scalars, multidimensional
25 * arrays, that read from the NetCDF file.
26 *
27 * The data read from NetCDF file are stored in a continuous memory, and the data layout is described
28 * by using a vector (shape). AMRex Box can be constructed using the data shape information, and MultiFab
29 * data array can be setup using the data that stored in the NDArray.
30 */
31template <typename DataType>
32struct NDArray
33{
34 using DType = typename std::remove_const<DataType>::type;
35
36 /** \brief constructor */
37 explicit NDArray (const std::string vname, const std::vector<MPI_Offset>& vshape)
38 : name{vname}, shape{vshape}, ref_counted{1}, owned{true} {
39 data = new DType [this->ndim()];
40 }
41
42 /** \brief default constructor */
43 NDArray () : name{"null"}, ref_counted{1}, owned{false}, data{nullptr} {}
44
45 /** \brief copy constructor */
46 NDArray (const NDArray& array) {
47 name = array.name;
48 shape = array.shape;
49 data = array.data;
50 owned = false;
51 ref_counted.fetch_add(1, std::memory_order_relaxed);
52 }
53
54 /** \brief copy assignment */
55 NDArray& operator=(const NDArray& array) {
56 name = array.name;
57 shape = array.shape;
58 data = array.data;
59 owned = false;
60 ref_counted.fetch_add(1, std::memory_order_relaxed);
61 return *this;
62 }
63
64 /** \brief destructor */
65 ~NDArray () {
66 ref_counted.fetch_sub(1, std::memory_order_acq_rel);
67 if (ref_counted == 1 && owned) delete[] data;
68 }
69
70 /** \brief get the data pointer */
71 decltype(auto) get_data () {
72 ref_counted.fetch_add(1, std::memory_order_relaxed);
73 return data;
74 }
75
76 /** \brief get the variable name */
77 std::string get_vname () {
78 return name;
79 }
80
81 /** \brief get the variable data shape */
82 std::vector<MPI_Offset> get_vshape () {
83 return shape;
84 }
85
86 /** \brief return the total number of data */
87 size_t ndim () {
88 size_t num = 1;
89 int isize = static_cast<int>(shape.size());
90 for (auto i=0; i<isize; ++i) num *= shape[i];
91 return num;
92 }
93
94 /** \brief set the data shape information */
95 void set_vshape (std::vector<MPI_Offset> vshape) {
96 shape = vshape;
97 }
98
99 private:
100 std::string name;
101 std::vector<MPI_Offset> shape;
102 mutable std::atomic<size_t> ref_counted;
103 bool owned;
104 DType* data;
105};

180/**
181 * \brief Helper function for reading data from NetCDF file into a
182 * provided FAB.
183 *
184 * @param iv Index for which variable we are going to fill
185 * @param nc_arrays Arrays of data from NetCDF file
186 * @param var_name Variable name
187 * @param NC_dim_type Dimension type for the variable as stored in the NetCDF file
188 * @param temp FAB where we store the variable data from the NetCDF Arrays
189 */
190template<class FAB,typename DType>
191void
192fill_fab_from_arrays (int iv,
193 amrex::Vector<NDArray<amrex::Real>>& nc_arrays,
194 const std::string& var_name,
195 NC_Data_Dims_Type& NC_dim_type,
196 FAB& temp)
197{
198 int nsx, nsy, nsz;
199 if (NC_dim_type == NC_Data_Dims_Type::SN_WE) {
200 nsz = 1;
201 nsy = nc_arrays[iv].get_vshape()[0];
202 nsx = nc_arrays[iv].get_vshape()[1];
203 // amrex::Print() << "TYPE SN WE " << nsy << " " << nsx << std::endl;
204 } else if (NC_dim_type == NC_Data_Dims_Type::BT_SN_WE) {
205 nsz = nc_arrays[iv].get_vshape()[0];
206 nsy = nc_arrays[iv].get_vshape()[1];
207 nsx = nc_arrays[iv].get_vshape()[2];
208 // amrex::Print() << "TYPE BT SN WE " << nz << " " << nsy << " " << nsx << std::endl;
209 } else if (NC_dim_type == NC_Data_Dims_Type::Time_SN_WE) {
210 nsz = 1;
211 nsy = nc_arrays[iv].get_vshape()[1];
212 nsx = nc_arrays[iv].get_vshape()[2];
213 // amrex::Print() << "TYPE TIME SN WE " << nsy << " " << nsx << std::endl;
214 } else if (NC_dim_type == NC_Data_Dims_Type::Time_BT_SN_WE) {
215 nsz = nc_arrays[iv].get_vshape()[1];
216 nsy = nc_arrays[iv].get_vshape()[2];
217 nsx = nc_arrays[iv].get_vshape()[3];
218 // amrex::Print() << "TYPE TIME BT SN WE " << nsx << " " << nsy << " " << nsz << std::endl;
219 } else {
220 amrex::Abort("Dont know this NC_Data_Dims_Type");
221 }
222
223 amrex::Box my_box;
224
225 if (var_name == "u" || var_name == "ubar" || var_name == "mask_u" || var_name == "x_u" ||
226 var_name == "y_u" || var_name == "sustr")
227 {
228 my_box.setSmall(amrex::IntVect(0,-1,0));
229 my_box.setBig(amrex::IntVect(nsx-1,nsy-2,nsz-1));
230 my_box.setType(amrex::IndexType(amrex::IntVect(1,0,0)));
231 }
232 else if (var_name == "v" || var_name == "vbar" || var_name == "mask_v" || var_name == "x_v" ||
233 var_name == "y_v" || var_name == "svstr")
234 {
235 my_box.setSmall(amrex::IntVect(-1,0,0));
236 my_box.setBig(amrex::IntVect(nsx-2,nsy-1,nsz-1));
237 my_box.setType(amrex::IndexType(amrex::IntVect(0,1,0)));
238 }
239 else if (var_name == "mask_psi" || var_name == "x_psi" || var_name == "y_psi")
240 {
241 my_box.setSmall(amrex::IntVect(0,0,0));
242 my_box.setBig(amrex::IntVect(nsx-1,nsy-1,nsz-1));
243 my_box.setType(amrex::IndexType(amrex::IntVect(1,1,0)));
244 }
245 else // everything cell-centered -- these have one ghost cell
246 {
247 my_box.setSmall(amrex::IntVect(-1,-1,0));
248 my_box.setBig(amrex::IntVect(nsx-2,nsy-2,nsz-1));
249 }
250
251 // amrex::Print() <<" NAME and BOX for this VAR " << var_name << " " << my_box << std::endl;
252
253#ifdef AMREX_USE_GPU
254 // Make sure temp lives on CPU since nc_arrays lives on CPU only
255 temp.resize(my_box,1,amrex::The_Pinned_Arena());
256#else
257 temp.resize(my_box,1);
258#endif
259 amrex::Array4<DType> fab_arr = temp.array();
260
261 int ioff = temp.box().smallEnd()[0];
262 int joff = temp.box().smallEnd()[1];
263
264 auto num_pts_in_box = my_box.numPts();
265 auto num_pts_in_src = nsx*nsy*nsz;
266 // amrex::Print() <<" NUM PTS IN BOX " << num_pts_in_box << std::endl;
267 // amrex::Print() <<" NUM PTS IN SRC " << num_pts_in_src << std::endl;
268 AMREX_ALWAYS_ASSERT(num_pts_in_box >= num_pts_in_src);
269
270 for (int n(0); n < num_pts_in_src; ++n) {
271 int nplane = nsx*nsy;
272 int k = n / nplane;
273 int j = (n - k*nplane) / nsx + joff;
274 int i = n - k*nplane - (j-joff) * nsx + ioff;
275 fab_arr(i,j,k,0) = static_cast<DType>(*(nc_arrays[iv].get_data()+n));
276 }
277}

279/**
280 * \brief Function to read NetCDF variables and fill the corresponding Array4's
281 *
282 * @param fname Name of the NetCDF file to be read
283 * @param nc_var_names Variable names in the NetCDF file
284 * @param NC_dim_types NetCDF data dimension types
285 * @param fab_vars Fab data we are to fill
286 * @param one_time Whether to read in a single time step from file
287 * @param fill_time What time step to read from file, if only reading one step
288 */
289template<class FAB,typename DType>
290void
291BuildFABsFromNetCDFFile (const amrex::Box& domain,
292 const std::string &fname,
293 amrex::Vector<std::string> nc_var_names,
294 amrex::Vector<enum NC_Data_Dims_Type> NC_dim_types,
295 amrex::Vector<FAB*> fab_vars,
296 bool one_time=false, int fill_time=0)
297{
298 int ioproc = amrex::ParallelDescriptor::IOProcessorNumber(); // I/O rank
299
300 amrex::Vector<NDArray<amrex::Real>> nc_arrays(nc_var_names.size());
301
302
303 ReadNetCDFFile(fname, nc_var_names, nc_arrays, one_time, fill_time); // this is filled only on proc 0
304
305 for (int iv = 0; iv < nc_var_names.size(); iv++)
306 {
307 FAB tmp;
308 if (amrex::ParallelDescriptor::IOProcessor()) {
309 fill_fab_from_arrays<FAB,DType>(iv, nc_arrays, nc_var_names[iv], NC_dim_types[iv], tmp);
310 }
311
312 int ncomp = tmp.nComp();
313 amrex::Box box = tmp.box();
314
315 amrex::ParallelDescriptor::Bcast(&box, 1, ioproc);
316 amrex::ParallelDescriptor::Bcast(&ncomp, 1, ioproc);
317
318 if (!amrex::ParallelDescriptor::IOProcessor()) {
319#ifdef AMREX_USE_GPU
320 tmp.resize(box,ncomp,amrex::The_Pinned_Arena());
321#else
322 tmp.resize(box,ncomp);
323#endif
324 }
325
326 amrex::ParallelDescriptor::Bcast(tmp.dataPtr(), tmp.size(), ioproc);
327
328 // Shift box by the domain lower corner
329 amrex::Box fab_bx = tmp.box();
330 amrex::Dim3 dom_lb = lbound(domain);
331 fab_bx += amrex::IntVect(dom_lb.x,dom_lb.y,dom_lb.z);
332 // fab_vars points to data on device
333 fab_vars[iv]->resize(fab_bx,1);
334#ifdef AMREX_USE_GPU
335 amrex::Gpu::copy(amrex::Gpu::hostToDevice,
336 tmp.dataPtr(), tmp.dataPtr() + tmp.size(),
337 fab_vars[iv]->dataPtr());
338#else
339 // Provided by BaseFab inheritance through FArrayBox
340 fab_vars[iv]->copy(tmp,tmp.box(),0,fab_bx,0,1);
341#endif
342 }
343}
