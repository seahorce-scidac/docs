49 //
50 // NDArray is the datatype designed to hold any data, including scalars, multidimensional
51 // arrays, that read from the NetCDF file.
52 //
53 // The data read from NetCDF file are stored in a continuous memory, and the data layout is described
54 // by using a vector (shape). AMRex Box can be constructed using the data shape information, and MultiFab
55 // data array can be setup using the data that stored in the NDArray.
56 //
57 template <typename DataType>
58 struct NDArray
59 {
60  using DType = typename std::remove_const<DataType>::type;
61 
62  // constructor
63  explicit NDArray (const std::string vname, const std::vector<size_t>& vshape)
64  : name{vname}, shape{vshape}, ref_counted{1}, owned{true} {
65  data = new DType [this->ndim()];
66  }
67 
68  // default constructor
69  NDArray() : name{"null"}, ref_counted{1}, owned{false}, data{nullptr} {}
70 
71  // copy constructor
72  NDArray(const NDArray& array) {
73  name = array.name;
74  shape = array.shape;
75  data = array.data;
76  owned = false;
77  ref_counted.fetch_add(1, std::memory_order_relaxed);
78  }
79 
80  // copy assignment
81  NDArray& operator=(const NDArray& array) {
82  name = array.name;
83  shape = array.shape;
84  data = array.data;
85  owned = false;
86  ref_counted.fetch_add(1, std::memory_order_relaxed);
87  return *this;
88  }
89 
90  // destructor
91  ~NDArray() {
92  ref_counted.fetch_sub(1, std::memory_order_acq_rel);
93  if (ref_counted == 1 && owned) delete[] data;
94  }
95 
96  // get the data pointer
97  decltype(auto) get_data() {
98  ref_counted.fetch_add(1, std::memory_order_relaxed);
99  return data;
100  }
101 
102  // get the variable name
103  std::string get_vname() {
104  return name;
105  }
106 
107  // get the variable data shape
108  std::vector<size_t> get_vshape() {
109  return shape;
110  }
111 
112  // return the total number of data
113  size_t ndim() {
114  size_t num = 1;
115  int isize = static_cast<int>(shape.size());
116  for (auto i=0; i<isize; ++i) num *= shape[i];
117  return num;
118  }
119 
120  // set the data shape information
121  void set_vshape(std::vector<size_t> vshape) {
122  shape = vshape;
123  }
124 
125  private:
126  std::string name;
127  std::vector<size_t> shape;
128  mutable std::atomic<size_t> ref_counted;
129  bool owned;
130  DType* data;
