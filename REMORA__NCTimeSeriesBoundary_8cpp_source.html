141void NCTimeSeriesBoundary::update_interpolated_to_time (amrex::Real time) {
142 // Figure out time index:
143 AMREX_ASSERT(time >= bry_times[0]);
144 AMREX_ASSERT(time <= bry_times[bry_times.size()-1]);
145 int i_time_before_old = i_time_before;
146 for (int nt=0; nt < bry_times.size()-1; nt++) {
147 if ((bry_times[nt] <= time) and (bry_times[nt+1] >= time)) {
148 i_time_before = nt;
149 time_before = bry_times[nt];
150 time_after = bry_times[nt+1];
151 break;
152 }
153 }
154
155 int i_time_after = i_time_before + 1;
156 if (i_time_before_old + 1 == i_time_before) {
157 // swap multifabs so we only have to read in one MultiFab
158 std::swap(xlo_dat_before, xlo_dat_after);
159 std::swap(xhi_dat_before, xhi_dat_after);
160 std::swap(ylo_dat_before, ylo_dat_after);
161 std::swap(yhi_dat_before, yhi_dat_after);
162 read_in_at_time(xlo_dat_after,xhi_dat_after, ylo_dat_after, yhi_dat_after, i_time_after);
163 } else if (i_time_before_old != i_time_before) {
164 read_in_at_time(xlo_dat_after,xhi_dat_after, ylo_dat_after, yhi_dat_after, i_time_after);
165 read_in_at_time(xlo_dat_before,xhi_dat_before, ylo_dat_before,yhi_dat_before, i_time_before);
166 }
167
168 amrex::Real dt = time_after - time_before;
169
170 amrex::Real time_before_copy = time_before;
171
172 amrex::Array4<amrex::Real> xlo_interp_arr = xlo_dat_interp.array();
173 amrex::Array4<amrex::Real> xhi_interp_arr = xhi_dat_interp.array();
174 amrex::Array4<amrex::Real> ylo_interp_arr = ylo_dat_interp.array();
175 amrex::Array4<amrex::Real> yhi_interp_arr = yhi_dat_interp.array();
176
177 amrex::Array4<amrex::Real> xlo_before_arr = xlo_dat_before.array();
178 amrex::Array4<amrex::Real> xhi_before_arr = xhi_dat_before.array();
179 amrex::Array4<amrex::Real> ylo_before_arr = ylo_dat_before.array();
180 amrex::Array4<amrex::Real> yhi_before_arr = yhi_dat_before.array();
181
182 amrex::Array4<amrex::Real> xlo_after_arr = xlo_dat_after.array();
183 amrex::Array4<amrex::Real> xhi_after_arr = xhi_dat_after.array();
184 amrex::Array4<amrex::Real> ylo_after_arr = ylo_dat_after.array();
185 amrex::Array4<amrex::Real> yhi_after_arr = yhi_dat_after.array();
186
187 amrex::ParallelFor(xlo_dat_interp.box(), [=] AMREX_GPU_DEVICE (int i, int j, int k)
188 {
189 xlo_interp_arr(i,j,k) = xlo_before_arr(i,j,k) + (time - time_before_copy) * (xlo_after_arr(i,j,k) - xlo_before_arr(i,j,k)) / dt;
190 });
191
192 amrex::ParallelFor(xhi_dat_interp.box(), [=] AMREX_GPU_DEVICE (int i, int j, int k)
193 {
194 xhi_interp_arr(i,j,k) = xhi_before_arr(i,j,k) + (time - time_before_copy) * (xhi_after_arr(i,j,k) - xhi_before_arr(i,j,k)) / dt;
195 });
196
197 amrex::ParallelFor(ylo_dat_interp.box(), [=] AMREX_GPU_DEVICE (int i, int j, int k)
198 {
199 ylo_interp_arr(i,j,k) = ylo_before_arr(i,j,k) + (time - time_before_copy) * (ylo_after_arr(i,j,k) - ylo_before_arr(i,j,k)) / dt;
200 });
201

205 });
