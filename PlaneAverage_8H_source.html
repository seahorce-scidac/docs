102 AMREX_FORCE_INLINE
103 amrex::Real
104 PlaneAverage::line_average_interpolated(amrex::Real x, int comp) const
105 {
106  AMREX_ALWAYS_ASSERT(comp >= 0 && comp < m_ncomp);
107 
108  amrex::Real c = 0.0;
109  int ind = 0;
110 
111  if (x > m_xlo + 0.5 * m_dx) {
112  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
113  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
114  c = (x - x1) / m_dx;
115  }
116 
117  if (ind + 1 >= m_ncell_line) {
118  ind = m_ncell_line - 2;
119  c = 1.0;
120  }
121 
122  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
123 
124  return m_line_average[m_ncomp * ind + comp] * (1.0 - c) +
125  m_line_average[m_ncomp * (ind + 1) + comp] * c;
126 }

175  amrex::Box pbx = PerpendicularBox<IndexSelector>(bx, amrex::IntVect{0, 0, 0});

177  ParallelFor(amrex::Gpu::KernelInfo().setReduction(true), pbx, [=]

183  amrex::Box lbx = ParallelBox<IndexSelector>(bx, amrex::IntVect{p_i, p_j, p_k});
